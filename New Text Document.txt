//operator overloading

#include <iostream>


using namespace std;
 
 
class complex 
{
    public:
    int real;
    int imag;
    


void display()
{
    cout<<real<<"+"<<imag<<"i"<<endl;
}

complex operator+(complex const&s1)
{
    complex sum;
    sum.real=real+s1.real;
    sum.imag=imag+s1.imag;
    return sum;
}
};
int main()
{
    complex c1;
    complex c2;
    c1.real=2;
    c1.imag=3;
    cout<<"The first complex number is:";
    c1.display();
    
    c2.real=4;
    c2.imag=5;
    cout<<"The second complex number is:";
    c2.display();
    cout<<"The addition of two complex numbers are:";
    complex c3=c1+c2;
    c3.display();
}
